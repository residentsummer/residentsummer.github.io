(defn time-range
  "Creates a vector of two Joda DateTime objects, representing a time range"
  ([start]
   (time-range start nil))
  ([start end]
   {:pre [(not-empty start)]}
   (mapv #(or (some->> (not-empty %)
                       (apply t/date-time))
              (t/now))
         [start end])))
